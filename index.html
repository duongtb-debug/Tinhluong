// HÀM HỖ TRỢ ĐỔI "2026-01" THÀNH "Tháng 01/2026" function formatPeriodVN(periodStr) { if (!periodStr) return "--"; if (periodStr.includes("-")) { let parts = periodStr.split("-"); return `Tháng ${parts[1]}/${parts[0]}`; } return periodStr; } // RENDER BẢNG CƯỚC VIỄN THÔNG (CÓ ĐỊNH DẠNG VIỆT NAM & NÚT SỬA) function renderTelecomTable() { let viewMonthMode = document.getElementById('telecom-view-mode').value; const tbody = document.getElementById('tbody-telecom'); tbody.innerHTML = ""; let sum = 0; let activeTelecoms = telecomRecordsAll.filter(r => { let rYear = r.period ? r.period.split("-")[0] : r.year || "2026"; if (rYear !== CURRENT_SELECTED_YEAR) return false; if (viewMonthMode === "ALL_YEAR") return true; return r.period && r.period.endsWith(`-${viewMonthMode}`); }); activeTelecoms.forEach((r, i) => { sum += Number(r.cost); let fileLinksHtml = ` Mở Folder Tháng`; if (r.fileList && r.fileList.length > 0) { fileLinksHtml += `
(${r.fileList.length} file đính kèm)
`; } // Định dạng tiếng Việt: Tháng 01/2026 let displayPeriod = formatPeriodVN(r.period); tbody.innerHTML += ` ${i+1} ${r.location || 'Tổng công ty Giấy Việt Nam'} ${r.provider} ${r.package} ${r.cid} ${Number(r.cost).toLocaleString('vi-VN')} VND ${displayPeriod} ${fileLinksHtml} `; }); document.getElementById('dash-pending-pay').innerText = sum.toLocaleString('vi-VN') + " VND"; } // HÀM ĐỔ DỮ LIỆU CƯỚC LÊN FORM ĐỂ SỬA function editTelecomRecord(id) { let item = telecomRecordsAll.find(d => d.id === id); if (!item) return; document.getElementById('telecom-edit-id').value = item.id; document.getElementById('tel-location').value = item.location || ''; document.getElementById('tel-provider').value = item.provider || 'VNPT'; document.getElementById('tel-package').value = item.package || ''; document.getElementById('tel-cid').value = item.cid || ''; document.getElementById('tel-cost').value = item.cost || 0; document.getElementById('tel-period').value = item.period || `${CURRENT_SELECTED_YEAR}-01`; document.getElementById('btn-telecom-text').innerText = "Cập Nhật Cước Viễn Thông"; window.scrollTo({ top: document.getElementById('telecom-main-form').offsetTop - 20, behavior: 'smooth' }); }
